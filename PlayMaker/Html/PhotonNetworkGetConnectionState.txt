	<div id="paramName">

	<div id="paramDesc">

<div id="paramRow">
	<div id="paramName">
Is Disconnected Event</div>
	<div id="paramDesc">

</div>

<div id="paramRow">
	<div id="paramName">
Is Disconnecting Event</div>
	<div id="paramDesc">
Event to send if Photon Network state is 'Disconnecting'</div>
</div>

<div id="paramRow">
	<div id="paramName">
Is Initializing Application Event</div>

</div>
